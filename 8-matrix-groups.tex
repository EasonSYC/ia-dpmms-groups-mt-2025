\section{Matrix Groups}
\label{sec:matrix-groups}

Let
\[
    \matrixset_n\para{\RR} = \brce{n \times n \text{ matrices with entries in } \RR}.
\]

Matrix multiplication is associative and the matrix
\[
    \identityM = \diag\para{1, 1, \ldots, 1} = \begin{pmatrix}
        1 &   &        & 0 \\
          & 1 &        &   \\
          &   & \ddots &   \\
        0 &   &        & 1
    \end{pmatrix}
\]
gives an identity. However, we know not all matrices are invertible.

\subsection{Linear Groups}
\label{sec:linear-groups}

\begin{lemma}[Invertible Matrices]
    \label{lem:invertible-matrices}%
    \(\vm{M} \in \matrixset_n\para{\RR}\) has an inverse if and only if \(\det \vm{M} \neq 0\).
\end{lemma}

\begin{definition}[General Linear Group]
    \label{def:general-linear}%
    Let
    \[
        \GL_n\para{\RR} = \set{\vm{M} \in \matrixset_n \para{\RR}}{\det \vm{M} \neq 0}
    \]
    is the \emph{general linear group of \(n \times n\) matrices over \(\RR\)}.
\end{definition}

\begin{lemma}[The Determinant is a Homomorphism]
    \label{lem:determinant}%
    For matrices \(\vm{A}, \vm{B} \in \matrixset_n\para{\RR}\), we have
    \[
        \det\para{\vm{A}, \vm{B}} = \det\vm{A} \det\vm{B}
    \]
    so \(\det\) is a homomorphism
    \[
        \function{\det}{\GL_n\para{\RR}}{\RR\multiplicative}{\vm{M}}{\det\vm{M}}.
    \]
\end{lemma}

\begin{definition}[Special Linear Group]
    \label{def:special-linear}%
    The subgroup
    \[
        \SL_n\para{\RR} = \ker \det = \set{\vm{M} \in \matrixset_n \para{\RR}}{\det \vm{M} = 1}
    \]
    is the \emph{special linear group of \(n \times n\) matrices over \(RR\)}.
\end{definition}

By \autoref{thm:first-isomorphism} (First Isomorphism Theorem), we have
\[
    \SL_n\para{\RR} \normalsub \GL_n\para{\RR}
\]
and
\[
    \GL_n\para{\RR} / \SL_n\para{\RR} \isomorphic \img\det.
\]

For any \(x \in \RR\multiplicative\), we have
\[
    \det \begin{pmatrix}
        x &   &        & 0 \\
          & 1 &        &   \\
          &   & \ddots &   \\
        0 &   &        & 1
    \end{pmatrix} = \det\diag\para{x, 1, \ldots, 1} = x
\]
so \(\det\) is surjective,
\[
    \img\det = \RR\multiplicative.
\]

Everything above makes sense as well when \(\RR\) is replaced by \(\CC\), so we may speak of \(\GL_n\para{\CC}\) and \(\SL_n\para{\CC}\), and similarly,
\[
    \GL_n\para{\CC} / \SL_n\para{\CC} \isomorphic \CC\multiplicative.
\]

\subsection{Change of Basis}
\label{sec:change-of-basis}

There is a natural action by conjugation \(\GL_n\para{\RR} \actson \matrixset_n\para{\RR}\) where
\[
    \vm{P}\para{\vm{A}} = \vm{P} \vm{A} \vm{P}\inverse
\]
for \(\vm{P} \in \GL_n\para{\RR}\) and \(\vm{A} \in \matrixset_n\para{\RR}\).

\begin{proposition}[Conjugation represent Change of Basis]
    \label{prop:conjugation-change-of-basis}%
    Let \(\mathcal{V}\) be an \(n\)-dimensional vector space over \(\RR\) and \(\functiondomain{\alpha}{\mathcal{V}}{\mathcal{V}}\) be a linear map/endomorphism (not necessarily injective/surjective). If \(\vm{A} \in \matrixset_n\para{\RR}\) represents \(\alpha\) with respect to some basis, then the orbit
    \[
        \orb_{\GL_n\para{\RR}} \para{\vm{A}} = \set{\vm{P} \vm{A} \vm{P}\inverse}{\vm{P} \in \GL_n\para{\RR}}
    \]
    consists of \emph{all} the matrices that represent \(\alpha\) in some basis.
\end{proposition}

\begin{proof}
    \begin{figure}[ht!]
        \centering
        \begin{tikzcd}
            \RR^n \arrow[r, "\phi", "\isomorphic"'] \arrow[d, "\vm{A}"']
            \arrow[rr, bend left, "\vm{P}"]
            & \mathcal{V} \arrow[d, "\alpha"] 
            & \RR^n \arrow[l, "\psi"', "\isomorphic"] \arrow[d, "\vm{B}"]\\
            \RR^n \arrow[r, "\phi", "\isomorphic"']
            \arrow[rr, bend right, "\vm{P}"']
            & \mathcal{V}
            & \RR^n \arrow[l, "\psi"', "\isomorphic"]
        \end{tikzcd}
        \caption{Change of Basis and Conjugation}
        \label{fig:change-of-basis-conjugation}
    \end{figure}

    \autoref{fig:change-of-basis-conjugation} shows an outline of the proof.

    A choice of basis \(\brce{\vm{v}_1, \vm{v}_2, \ldots, \vm{v}_n}\) for \(\mathcal{V}\) defines an isomorphism of vector spaces
    \[
        \function{\phi}{\RR^n}{\mathcal{V}}{\para{\lambda_1, \lambda_2, \ldots, \lambda_n}\transpose}{\sum_{i = 1}^{n} \lambda_i \vm{v}_i}.
    \]

    This claims that \(\vm{A}\) represents \(\alpha\) in this basis means that \(\alpha = \phi \vm{A} \phi\inverse\).

    Likewise, another basis \(\brce{\vm{u}_1, \vm{u}_2, \ldots, \vm{u}_n}\) for \(\mathcal{V}\) defines an isomorphism
    \[
        \functiondomain{\psi}{\RR^n}{\mathcal{V}}
    \]
    and a matrix \(\vm{B}\) represents \(\alpha\) in the coordinates
    \[
        \alpha = \psi \vm{B} \psi\inverse.
    \]

    Therefore,
    \begin{align*}
        \vm{B} &= \psi\inverse \alpha \psi \\
        &= \para{\psi\inverse \circ \phi} \vm{A} \para{\phi\inverse \circ \psi} \\
        &= \vm{P} \vm{A} \vm{P}\inverse
    \end{align*}
    for some \(\vm{P} \in \GL_n\para{\RR}\) representing
    \[
        \functiondomain{\psi\inverse \circ \phi}{\RR^n}{\RR^n}
    \]
    since this mapping is invertible. Thus, the set of all matrices representing \(\alpha\) is indeed contained in the orbit.

    Conversely, if \(\vm{B} = \vm{P} \vm{A} \vm{P}\inverse\) for some \(\vm{P} \in \GL_n\para{\RR}\), then let
    \[
        \functiondomain{\psi = \phi \circ \vm{P}\inverse}{\RR^n}{\mathcal{V}},
    \]
    we get a basis
    \[
        \brce{\vm{u}_i = \phi\para{\unitv{e}_i}}
    \]
    where \(\unitv{e}_i\) is the standard basis for \(\RR^n\). In this basis, \(\vm{B}\) represents \(\alpha\), as desired.
\end{proof}

\subsection{M\"{o}bius Transformations Revisited}
\label{sec:mobius-transformations-revisited}

Recall that multiplication in \(\mobius\) looks similar to multiplication of \(2 \times 2\) complex matrices.

\begin{proposition}[Identification of \(\mobius\)]
    \label{prop:identification-of-mobius-group}%
    Identity
    \[
        \CC\multiplicative = \set{\begin{pmatrix}
            \lambda & 0 \\ 0 & \lambda
        \end{pmatrix}}{\lambda \in \CC \setminus \brce{0}} \subgroup \GL_2\para{\CC}.
    \]

    Then \(\CC\multiplicative \normalsub \GL_2\para{\CC}\), and
    \[
        \mobius \isomorphic \GL_2\para{\CC} / \CC\multiplicative.
    \]
\end{proposition}

\begin{proof}
    Define the na\"{i}ve map
    \[
        \function{\phi}{\GL_2\para{\CC}}{\mobius}{
            \begin{pmatrix}
                a & b \\ c & d
            \end{pmatrix}}{\para{z \mapsto \frac{az + b}{cz + d}}}.
    \]

    By computation of multiplication in \(\mobius\), we see that \(\phi\) is a homomorphism and is clearly surjective.

    A matrix \(\begin{pmatrix}
        a & b \\ c & d
    \end{pmatrix} \in \ker \phi\) if and only if its image fixes \(0\), \(1\) and \(\infty\), if and only if
    \[
        b = c = 0, a = d.
    \]

    So
    \[
        \ker \phi = \CC\multiplicative \normalsub \GL_2\para{\CC}.
    \]

    The result then directly follows from \autoref{thm:first-isomorphism} (First Isomorphism Theorem).
\end{proof}

\subsection{Orthogonal Groups}
\label{sec:orthogonal-groups}

We write \(\norm{\cdot}\) for the usual notion of length in \(\RR^n\): let
\[
    \norm{\vm{u}} = \sqrt{\sum_{i = 1}^{n} u_i^2}
\]
be the \emph{Euclidean norm}.

\begin{definition}[Orthogonal Group]
    \label{def:orthogonal-group}%
    The \emph{\(n\)-dimensional orthogonal group} is the subgroup of \(\GL_r\para{\RR}\) that preserve distance in \(\RR^n\):
    \[
        \Or\para{n} = \set{\vm{M} \in \GL_n\para{\RR}}{\forall \vm{v} \in \RR^n: \norm{\vm{Mv}} = \norm{\vm{v}}}.
    \]
\end{definition}

In fact, the \emph{dot product}
\[
    \vm{u} \cdot \vm{v} = \sum_{i = 1}^{n} u_i v_i
\]
is often more convenient to work with. In fact, preserving the dot product is equivalent to preserving lengths.

\begin{lemma}[Polarisation Identity]
    \label{lem:polarisation-identity}%
    For \(\vm{u}, \vm{v} \in \RR^n\), we have
    \[
        2 \para{\vm{u} \cdot \vm{v}} = \norm{\vm{u}}^2 + \norm{\vm{v}}^2 - \norm{\vm{u} - \vm{v}}^2.
    \]
\end{lemma}

\begin{proof}
    We have
    \begin{align*}
        \norm{\vm{u} - \vm{v}}^2 &= \para{\vm{u} - \vm{v}} \cdot \para{\vm{u} - \vm{v}} \\
        &= \norm{\vm{u}}^2 - 2 \para{\vm{u} \cdot \vm{v}} + \norm{\vm{v}}^2.
    \end{align*}
\end{proof}

We can characterise \(\Or\para{n}\) using the dot product.

\begin{lemma}[\(\Or\para{n}\) are Matrices Preserving the Dot Product]
    \label{lem:orthogonal-dot-product}%
    Orthogonal matrices are precisely those matrices preserving the dot product, i.e.,
    \[
        \Or\para{n} = \set{\vm{M} \in \GL_n\para{\RR}}{\forall \vm{x}, \vm{y} \in \RR^n: \para{\vm{Mx}} \cdot \para{\vm{My}} = \vm{x} \cdot \vm{y}}.
    \]

    Equivalently, for \(\vm{M} \in \GL_{n} \para{\RR}\), \(\vm{M} \in \Or\para{n}\) if and only if \(\vm{M}\) preserves dot product.
\end{lemma}

\begin{proof}
    \(\impliedby\). If \(\para{\vm{Mx}} \cdot \para{\vm{My}} = \vm{x} \cdot \vm{y}\) for all \(\vm{x}, \vm{y} \in \RR^n\), then for any \(\vm{v} \in \RR^n\), we have
    \begin{align*}
        \norm{\vm{Mv}}^2 &= \para{\vm{Mv}} \cdot \para{\vm{Mv}} \\
        &=\vm{v} \cdot \vm{v} \\
        &= \norm{\vm{v}}^2,
    \end{align*}
    so \(\norm{\vm{Mv}} = \norm{\vm{v}}\), and hence \(\vm{M} \in \Or\para{n}\).

    \(\implies\). Conversely, for any \(\vm{M} \in \Or\para{n}\) and \(\vm{x}, \vm{y} \in \RR^n\), we have
    \begin{align*}
        2 \para{\vm{Mx} \cdot \vm{My}} &= \norm{\vm{Mx}}^2 + \norm{\vm{My}}^2 - \norm{\vm{Mx} - \vm{My}}^2 && \text{by \autoref{lem:polarisation-identity} (Polarisation Identity)} \\
        &= \norm{\vm{Mx}}^2 + \norm{\vm{My}}^2 - \norm{\vm{M} \para{\vm{x} - \vm{y}}}^2 \\
        &= \norm{\vm{x}}^2 + \norm{\vm{y}}^2 - \norm{\vm{x} - \vm{y}}^2 && \vm{M} \in \Or\para{n} \\
        &= 2 \para{\vm{x} \cdot \vm{y}}. && \text{by \autoref{lem:polarisation-identity} (Polarisation Identity)}
    \end{align*}

    Therefore, \(\para{\vm{Mx}} \cdot \para{\vm{My}} = \vm{x} \cdot \vm{y}\), and \(\vm{M}\) preserves dot product as desired.
\end{proof}

This quickly leads to a nice characterisation of matrices in \(\Or\para{n}\).

\begin{lemma}[Matrices in \(\Or\para{n}\)]
    \label{lem:orthogonal-matrices-orthonormal-transpose}%
    For \(\vm{M} \in \matrixset_n \para{\RR}\). The following are equivalent:
    \begin{enumerate}
        \item \(\vm{M} \in \Or\para{n}\);
        \item The columns of \(\vm{M}\) form an orthonormal basis;
        \item \(\vm{M}\transpose \vm{M} = \identityM\).
    \end{enumerate}
\end{lemma}

\begin{proof}
    \emph{(i) \(\implies\) (ii).} Let \(\brce{\unitv{e}_1, \unitv{e}_2, \ldots, \unitv{e}_n}\) be the standard basis for \(\RR^n\). The \(i\)-th column of \(\vm{M}\) is the column vector \(\vm{M} \unitv{e}_i\), since
    \[
        \para{\vm{M} \unitv{e}_i} \cdot \para{\vm{M} \unitv{e}_j} = \unitv{e}_i \cdot \unitv{e}_j = \kronecker_{ij} = \begin{cases}
            1, & i = j, \\
            0, & i \neq j.
        \end{cases}
    \]

    Therefore, the columns form an orthonormal basis indeed.

    \emph{(ii) \(\implies\) (iii).} As explained above, (ii) is equivalent to
    \[
        \para{\vm{M} \unitv{e}_i} \cdot \para{\vm{M} \unitv{e}_j }= \kronecker_{ij}.
    \]

    Since \(\vm{u} \cdot \vm{v} = \vm{u} \transpose \vm{v}\), this means
    \[
        \para{\vm{M} \unitv{e}_i}\transpose \para{\vm{M} \unitv{e}_j} = \kronecker_{ij}.
    \]
    
    The left-hand side satisfies
    \begin{align*}
        \para{\vm{M} \unitv{e}_i}\transpose \para{\vm{M} \unitv{e}_j} &= \unitv{e}_i\transpose \vm{M}\transpose \vm{M} \unitv{e}_j \\
        &= \para{\vm{M}\transpose \vm{M}}_{ij},
    \end{align*}
    since for any square matrix \(\vm{N}\), \(\unitv{e}_i \transpose \vm{N} \unitv{e}_j = \para{\vm{N}}_{ij}\) gives the \(\para{i, j}\)-th element of \(\vm{N}\). Therefore,
    \[
        \vm{M}\transpose \vm{M} = \identityM.
    \]

    \emph{(iii) \(\implies\) (i).} Suppose \(\vm{u}, \vm{v} \in \RR^n\), then
    \begin{align*}
        \para{\vm{Mu}} \cdot \para{\vm{Mv}} &= \para{\vm{Mu}}\transpose \para{\vm{Mv}} \\
        &= \vm{u}\transpose \vm{M}\transpose \vm{M} \vm{v} \\
        &= \vm{u}\transpose \identityM \vm{v} \\
        &= \vm{u}\transpose \vm{v} \\
        &= \vm{u} \cdot \vm{v},
    \end{align*}
    so we have \(\vm{M} \in \Or\para{n}\) by \autoref{lem:orthogonal-dot-product}.
\end{proof}

Recall that for any matrix \(\vm{M} \in \matrixset_{n} \para{\RR}\), we have \(\det \para{\vm{M}\transpose} = \det \para{\vm{M}}\). Hence, for any \(\vm{M} \in \Or\para{n}\),
\begin{align*}
    1 &= \det \identityM \\
    &= \det \para{\vm{M}\transpose\vm{M}} \\
    &= \para{\det \vm{M}\transpose} \para{\det \vm{M}} \\
    &= \para{\det \vm{M}}^2,
\end{align*}
and hence \(\det \vm{M} = \pm 1\).

\begin{definition}[Special Orthogonal Group]
    \label{def:special-orthogonal-group}%
    Consider the restriction of \(\det\) on \(\Or\para{n}\), giving
    \[
        \functiondomain{\rightbar{\det}_{\Or\para{n}}}{\Or\para{n}}{C_2 \isomorphic \brce{\pm 1}}.
    \]

    Define the \emph{\(n\)-dimensional special orthogonal group} as
    \[
        \SO\para{n} = \ker \rightbar{\det}_{\Or\para{n}} = \Or\para{n} \cap \SL_n\para{\RR}.
    \]
\end{definition}

Note that such restriction of the determinant is surjective, hence by \autoref{thm:first-isomorphism} (First Isomorphism Theorem), we have
\[
    \abs{\Or\para{n} : \SO\para{n}} = \abs{\ker \rightbar{\det}_{\Or\para{n}}} = 2.
\]

Recall that a positive determinant preserved orientation, and a negative determinant reverses orientation. Examples of elements of \(\Or\para{n} \setminus \SO\para{n}\) are provided by \emph{reflections}.

\begin{definition}[Reflections]
    \label{def:reflections}%
    Any \(\vm{v} \in \RR^n \setminus \brce{\vm{0}}\) defines an \emph{orthogonal plane}
    \[
        \vm{v}\perppl = P_\vm{v} = \set{\vm{u} \in \RR^n}{\vm{u} \cdot \vm{v} = 0}.
    \]

    The \emph{reflection} in \(P_\vm{v}\) is defined to be
    \[
        S_{\vm{v}} \para{\vm{x}} = \vm{x} - \frac{2 \para{\vm{x} \cdot \vm{v}}}{\norm{\vm{v}}^2} \vm{v}.
    \]
\end{definition}

    \label{fig:reflection-in-plane}
\end{figure}

\begin{remarks}
    \begin{enumerate}
        \item We sometimes use \(S_P\) for the reflection in the plane \(P\).
        \item If we replace \(\vm{v}\) by \(\frac{\vm{v}}{\norm{\vm{v}}}\), and assume \(\norm{\vm{v}} = 1\) (i.e. use the unit vector in \(\vm{v}\) direction, \(\unitv{v}\)), then \(P_{\vm{v}} = P_{\unitv{v}}\), and we have
        \[
        S_{\vm{v}} \para{\vm{x}} = \vm{x} - 2 \para{\vm{x} \cdot \vm{v}} \vm{v}.
        \]
    \end{enumerate}
\end{remarks}

We study some properties on those reflections.

\begin{lemma}[Reflections are Orthogonal and have Order 2]
    \label{lem:reflections-orthogonal-order-2}%
    For any \(\vm{v} \in \RR^n\), we have \(S_{\vm{v}}^2 = \identityM\), and \(S_{\vm{v}} \in \Or\para{n}\).    
\end{lemma}

\begin{proof}
    Assume \(\norm{\vm{v}} = 1\). Note that \(S_{\vm{v}}\) is indeed linear, so
    \[
        S_{\vm{v}} \in \matrixset_n \para{\RR}.
    \]

    Now, we have for any \(\vm{x} \in \RR^n\), we have
    \begin{align*}
        S_{\vm{v}} \para{\vm{x}} \cdot \vm{v} &= \para{\vm{x} \cdot \vm{v}} - 2 \para{\vm{x} \cdot \vm{v}} \cdot \para{\vm{v} \cdot \vm{v}} \\
        &= - \para{\vm{x} \cdot \vm{v}},
    \end{align*}
    and hence
    \begin{align*}
        S_{\vm{v}}^2 \para{\vm{x}} &= S_{\vm{v}} \para{\vm{x}} - 2 \para{S_{\vm{v}} \para{\vm{x}} \cdot \vm{v}} \vm{v} \\
        &= \vm{x} - 2\para{\vm{x} \cdot \vm{v}} \vm{v} + 2 \para{\vm{x} \cdot \vm{v}} \vm{v} \\
        &= \vm{x},
    \end{align*}
    and hence \(S_{\vm{v}}^2 = \identityM\). Hence, we see \(S_{\vm{v}}\) is non-singular, and hence
    \[
        S_{\vm{v}} \in \GL_n \para{\RR}.
    \]

    Finally, we verify that reflections indeed preserve lengths. For any \(\vm{x} \in \RR^n\),
    \begin{align*}
        \norm{S_{\vm{v}} \para{\vm{x}}}^2 &= \para{\vm{x} - 2 \para{\vm{x} \cdot \vm{v}} \vm{v}} \cdot \para{\vm{x} - 2 \para{\vm{x} \cdot \vm{v}} \vm{v}} \\
        &= \norm{\vm{x}}^2 - 4 \para{\vm{x} \cdot \vm{v}}^2 + 4 \para{\vm{x} \cdot \vm{v}}^2 \\
        &= \norm{\vm{x}}^2,
    \end{align*}
    and hence \(S_{\vm{v}} \in \Or\para{n}\) as desired.
\end{proof}

\begin{remark}
    To see that reflections are in \(\SO\para{n} \setminus \Or\para{n}\) indeed, let \(\norm{\vm{v}} = 1\), and we pick an orthonormal basis
    \[
        \brce{\vm{v}_1, \vm{v}_2, \ldots, \vm{v}_{n - 1}}
    \]
    for the \(\para{n-1}\)-dimensional subspace \(P_{\vm{v}}\). Then, in the basis
    \[
        \brce{\vm{v}_1, \vm{v}_2, \ldots, \vm{v}_{n - 1}, \vm{v}}
    \]
    for \(\RR^n\), \(S_{\vm{v}}\) is represented by the matrix
    \[
        \begin{pmatrix}
            1 &   &        &   & 0 \\
              & 1 &        &   &   \\
              &   & \ddots &   &   \\
              &   &        & 1 &   \\
            0 &   &        &   & -1
        \end{pmatrix} = \diag\para{1, 1, \ldots, 1, -1}.
    \]

    Therefore, \(\det S_{\vm{v}} = -1\), and \(S_{\vm{v}} \notin \SO\para{n}\).
\end{remark}

Reflections are in general not the only elements of \(\SO\para{n} \setminus \Or\para{n}\), specifically for \(n \geq 3\). A significant reason we study those reflections is they provide a useful generating set for \(\Or\para{n}\). Even stronger, we have \autoref{thm:reflections-generate}.

\begin{theorem}[Reflections Generate \(\Or\para{n}\)]
    \label{thm:reflections-generate}%
    Every \(\vm{A} \in \Or\para{n}\) is a product of at most \(n\) reflections.
\end{theorem}

\begin{proof}
    We proceed by induction on \(n\).

    \emph{Base Case.} For \(n = 1\), \(\Or\para{1} = \brce{\pm 1} = \ang{S_1} \isomorphic C_2\) and every element is indeed a product of at most \(1\) reflection. Note here \(S_1 \in \Or\para{1} \setminus \SO\para{1}\) is the reflection in the plane perpendicular to \(\para{-1}\), not the trivial group.

    \emph{Induction Step.} Let \(\brce{\unitv{e}_1, \unitv{e}_2, \ldots, \unitv{e}_n}\) be the standard basis for \(\RR^n\). Let
    \[
        \vm{v} = \unitv{e}_n - \vm{A} \unitv{e}_n,
    \]
    then \(\para{S_{\vm{v}} \vm{A}} \unitv{e}_n = \unitv{e}_n\), as in \autoref{fig:choice-of-v-for-on}.

    \begin{figure}[ht!]
        \centering
        \caption{Choice of \(\vm{v}\) for \(\vm{M} \in \Or\para{n}\)}
        \label{fig:choice-of-v-for-on}
    \end{figure}

    Since \(\para{S_{\vm{v}} \vm{A}}\) is orthogonal, \(\para{S_{\vm{v}} \vm{A}}\) preserves \(P_{\unitv{e}_n} = \RR^{n - 1} \times \brce{\vm{0}}\).

    By induction hypothesis, there exists
    \[
        \vm{v}_1, \vm{v}_2, \ldots, \vm{v}_{n - 1} \in \RR^{n - 1}
    \]
    such that when restricted to \(P_{\unitv{e}_n} = \RR^{n - 1} \times \brce{\vm{0}}\),
    \[
        S_{\vm{v}} \vm{A} = S_{\vm{v}_1} S_{\vm{v}_2} \cdots S_{\vm{v}_{n - 1}}
    \]
    and since both sides fix \(\vm{e}_n \notin P_{\unitv{e}_n}\), they agree on the whole of \(\RR^n\) as well. Hence,
    \[
        S_{\vm{v}} \vm{A} = S_{\vm{v}_1} S_{\vm{v}_2} \cdots S_{\vm{v}_{n - 1}}
    \]
    and hence
    \[
        \vm{A} = S_{\vm{v}} S_{\vm{v}_1} S_{\vm{v}_2} \cdots S_{\vm{v}_{n - 1}}
    \]
\end{proof}

We next aim to characterise elements of \(\Or\para{n}\) and \(\SO\para{n}\) in lower dimensions, specifically for \(\Or\para{2}\), \(\SO\para{2}\) and \(\SO\para{3}\).

\begin{lemma}[Elements in \(\Or\para{2}\)]
    \label{lem:elements-in-o2}%
    Let \(\vm{M} \in \Or\para{2}\).
    \begin{enumerate}
        \item If \(\vm{M} \notin \SO\para{2}\), then \(\vm{M}\) is a reflection.
        \item If \(\vm{M} \in \SO\para{2}\), then \(\vm{M}\) is a rotation about \(O\).
    \end{enumerate}
\end{lemma}

\begin{proof}
    Recall that \(\det S_{\vm{v}} = -1\), so
    \[
        \det \para{S_{\vm{v}_1} S_{\vm{v}_2} \cdots S_{\vm{v}_k}} = \para{-1}^k.
    \]

    By \autoref{thm:reflections-generate}, we may take \(k \leq 2\).
    \begin{enumerate}
        \item If \(\vm{M} \notin \SO\para{2}\), then \(k\) is odd, and hence \(k = 1\) giving \(\vm{M}\) is a reflection.
        \item If \(\vm{M} \in \SO\para{2}\), then \(k\) is even, so unless \(\vm{M} = \identityM\), then 
        \[
            \vm{M} = S_{\vm{u}} S_{\vm{v}}
        \]
        for some \(\vm{u}, \vm{v} \in \RR^2\) not parallel.

        Now we claim that \(\vm{M}\) only fixes the origin and hence is a rotation. Suppose for contradiction, that for some \(\vm{x} \neq \vm{0}\), we have that \(\vm{M}\) fixes \(\vm{x}\), note that
        \[
            S_{\vm{u}} S_{\vm{v}} \vm{x} = \vm{x} \iff S_{\vm{u}} \vm{x} = S_{\vm{v}} \vm{x}.
        \]

        \(\vm{v}\) is parallel to \(\vm{x} - S_{\vm{v}} \vm{x}\), and \(\vm{u}\) is parallel to \(\vm{x} - S_{\vm{u}} \vm{x}\). But since \(\vm{x} \neq \vm{0}\), we have \(\vm{x} \neq S_{\vm{v}} \vm{x} = S_{\vm{u}} \vm{x}\), and hence \(\vm{u}\) is parallel to \(\vm{v}\), which is not possible.

        Therefore, \(\vm{M}\) only fixes the origin \(O\), as desired.
    \end{enumerate}
\end{proof}

\begin{remark}
    Let \(\vm{M} \in \SO\para{2}\), since the columns form an orthonormal basis, we have
    \[
        \vm{M} = \begin{pmatrix}
            a & b \\ -b & a
        \end{pmatrix}
    \]
    with \(a, b \in \RR\) and \(a^2 + b^2 = \det \vm{M} = 1\). Then, let \(a = \cos \theta, b = \sin \theta\) for some \(\theta \in \RR\), and hence we have
    \[
        \vm{M} = \begin{pmatrix}
            \cos \theta & \sin \theta \\ - \sin \theta & \cos \theta
        \end{pmatrix}.
    \]

    Similarly, if \(\vm{M} \in \Or\para{2} \setminus \SO\para{2}\) is some reflection, we have
    \[
        \vm{M} = \begin{pmatrix}
            a & b \\ b & -a
        \end{pmatrix}
    \]
    with \(a, b \in \RR\) and \(-a^2 - b^2 = \det \vm{M} = -1\). Then, let \(a = \cos \theta, b = \sin \theta\) for some \(\theta \in \RR\), and hence we have
    \[
        \vm{M} = \begin{pmatrix}
            \cos \theta & \sin \theta \\ \sin \theta & -\cos \theta
        \end{pmatrix}.
    \]
\end{remark}

\begin{lemma}[Elements in \(\SO\para{3}\)]
    \label{lem:elements-in-so3}%
    If \(\vm{M} \in \SO\para{3}\), then \(\vm{M}\) is a rotation.
\end{lemma}

\begin{proof}
    Similarly to the proof of \autoref{lem:elements-in-o2}, \(\vm{M}\) is a product of at most \(3\) reflections. So either \(\vm{M} = \identityM\), or \(\vm{M} = S_{\vm{u}} S_{\vm{v}}\) for some non-parallel \(\vm{u}, \vm{v} \in \RR^3\).

    Since \(n = 3\), we have
    \[
        P_{\vm{u}} \cap P_{\vm{v}} = l
    \]
    for a line \(l\), so \(A\) fixes \(l\).

    Also, proceeding similarly as in \autoref{lem:elements-in-o2} shows us either \(S_{\vm{u}} = S_{\vm{v}}\) giving \(\vm{M} = \identityM\) or the fixed point is in \(P_{\vm{u}} \cap P_{\vm{v}} = l\).

    Therefore, any non-trivial \(\vm{M}\) only fixes a line \(l\), hence a rotation.
\end{proof}
