\section*{Schedules}
\label{sec:schedules}

\scheduleSection{Examples of groups}{4}{
Axioms for groups. Examples from geometry: symmetry groups of regular polygons, cube, tetrahedron. Permutations on a set; the symmetric group. Subgroups and homomorphisms. Symmetry groups as subgroups of general permutation groups. The M\"{o}bius group; cross-ratios, preservation of circles, the point at infinity. Conjugation. Fixed points of M\"{o}bius maps and iteration.}

\scheduleSection{Lagrange's theorem}{5}{
Cosets. Lagrange's theorem. Groups of small order (up to order 8). Quaternions. Fermat-Euler theorem from the group-theoretic point of view.}

\scheduleSection{Group actions}{4}{
Group actions; orbits and stabilizers. Orbit-stabilizer theorem. Cayley's theorem (every group is isomorphic to a subgroup of a permutation group). Conjugacy classes. Cauchy's theorem.}

\scheduleSection{Quotient groups}{4}{Normal subgroups, quotient groups and the isomorphism theorem.}

\scheduleSection{Matrix groups}{3}{The general and special linear groups; relation with the M\"{o}bius group. The orthogonal and special orthogonal groups. Proof (in \(\RR^3\)) that every element of the orthogonal group is the product of reflections and every rotation in \(\RR^3\) has an axis. Basis change as an example of conjugation.}

\scheduleSection{Permutations}{4}{Permutations, cycles and transpositions. The sign of a permutation. Conjugacy in \(S_n\) and in \(A_n\). Simple groups; simplicity of \(A_5\).}
