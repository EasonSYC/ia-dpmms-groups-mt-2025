\subsection{Introduction to the Algebraic Standpoint}
\label{sec:algebraic-standpoint}

We first define what it means to have a binary operation on a set -- this forms the foundation for algebra.

\begin{definition}[Binary Operation]
    \label{def:binary-operation}%
    A \emph{binary operation} on a set \(X\) is a function
    \[
        \cdot : X \times X \to X
    \]
    where \(X \times X\) is the \emph{Cartesian product} given by
    \[
        X \times X = \set{\para{x_1, x_2}}{x_1, x_2 \in X}.
    \]
    
    We denote
    \[
        \para{x_1, x_2} \mapsto x_1 \cdot x_2.
    \]
\end{definition}

Now we are ready to define a group as an algebraic structure, using the following axioms.

\begin{definition}[Group]
    \label{def:group}%
    A \emph{group} is a triple
    \[
        \para{G, \cdot, e}
    \]
    where
    \begin{itemize}
        \item \(G\) is a set,
        \item \(\cdot\) is a binary operation on \(G\), and
        \item \(e\) is an element of \(G\), denoted as \(e \in G\),
    \end{itemize}
    such that the following 4 axioms are satisfied:
    \begin{enumerate}
        \item \emph{Closure}. For all \(a, b \in G\), \(a \cdot b \in G\).
        \item \emph{Associativity}. For all \(a, b, c \in G\), \(
        \para{\para{a \cdot b} \cdot c} = \para{a \cdot \para{b \cdot c}}\).
        \item \emph{(Right) Identity}. For all \(a \in G\), \(a \cdot e = a\).
        \item \emph{(Right) Inverses}. For all \(a \in G\), there exists some \(b \in G\), such that \(a \cdot b = e\).
    \end{enumerate}
\end{definition}

\begin{remark}[Remark on Closure]
    The closure axiom is redundant, since \(\cdot\) being a binary operation means that \(G\) has to be closed under \(\cdot\)
\end{remark}

A group is an \textbf{algebraic structure with one operation}.

\begin{examples}
    \begin{itemize}
        \item \emph{Equilateral triangle.} The symmetries of an equilateral triangle forms a group, with respect to composition, and the identity symmetry is the identity element.
        \item \emph{Integers.} \(\para{\ZZ, +, 0}\) forms a group.
    \end{itemize}
\end{examples}

Some authors define the identity to be a two-sided identity, and the inverse to be a two-sided inverse. Indeed, this must be the case, and the identity and the inverse of an element must be unique, as in \autoref{prop:inverse-identity}.

\begin{proposition}[Properties on Inverses and Identities]
    \label{prop:inverse-identity}%
    Let \(\para{G, \cdot, e}\) be a group. Let \(a, b, b', e' \in G\).

    \begin{enumerate}
        \item \emph{Right-inverses are also left-inverses.} If \(a \cdot b = e\), then \(b \cdot a = e\).
        \item \emph{A right-identity is also a left-identity.} \(e \cdot a = a\).
        \item \emph{Inverses are unique.} If \(a, b, b'\) are such that \(a \cdot b = e\) and \(a \cdot b' = e\), then \(b = b'\).
        \item \emph{The identity is unique.} If \(a \cdot e' = a\), then \(e' = e\).
    \end{enumerate}
\end{proposition}

\begin{proof}
    \begin{enumerate}
        \item We have
        \begin{align*}
            b &= b \cdot e && \text{Identity Axiom} \\
            &= b \cdot \para{a \cdot b} && \text{Assumption} \\
            &= \para{b \cdot a} \cdot b. && \text{Associativity Axiom}
        \end{align*}

        By the Inverse Axiom, there exists \(c \in G\) such that \(b \cdot c = e\).

        We now right-multiply both sides by \(c\), giving
        \begin{align*}
            e &= b \cdot c && \text{Assumption on }c \\
            &= \para{\para{b \cdot a} \cdot b} \cdot c \\
            &= \para{b \cdot a} \cdot \para{b \cdot c} && \text{Associativity Axiom} \\
            &= \para{b \cdot a} \cdot e && \text{Assumption on }c \\
            &= b \cdot a. && \text{Identity Axiom}
        \end{align*}

        \item By the Inverse Axiom and the first part, there exists \(b \in G\) such that \(a \cdot b = b \cdot a = e\).
        
        Now, we have
        \begin{align*}
            e \cdot a &= \para{a \cdot b} \cdot a && \text{Assumption} \\
            &= a \cdot \para{b \cdot a} && \text{Associativity Axiom} \\
            &= a \cdot e && \text{Assumption} \\
            &= a. && \text{Identity Axiom}
        \end{align*}

        \item We have from the previous parts, that
        \begin{align*}
            b' &= e \cdot b' && \text{\autoref{prop:inverse-identity}, Part 2} \\
            &= \para{b \cdot a} \cdot b' && \text{\autoref{prop:inverse-identity}, Part 1, and Assumption} \\
            &= b \cdot \para{a \cdot b'} && \text{Associativity Axiom} \\
            &= b \cdot e && \text{Assumption} \\
            &= b && \text{Identity Axiom}.
        \end{align*}

        \item By the Inverse Axiom, and \autoref{prop:inverse-identity} Part 1, there exists \(b \in G\) such that \(b \cdot a = e\). We are given that \(a \cdot e' = a\), and hence \(b \cdot \para{a \cdot e'} = b \cdot a\).

        Hence,
        \begin{align*}
            e &= \para{b \cdot a} \cdot e' && \text{Associativity Axiom} \\
            &= e \cdot e' \\
            &= e' && \text{\autoref{prop:inverse-identity} Part 2}.
        \end{align*}
    \end{enumerate}    
\end{proof}

Part 3 of \autoref{prop:inverse-identity} tells us that inverses are unique.

\begin{notation}
    If \(a \cdot b = e\), we write \(b = a \inverse\).
\end{notation}

Part 1 of the same proposition tells us that
\[
    a \cdot a \inverse = a \inverse \cdot a = e
\]
which in turn means that
\[
    \para{a \inverse} \inverse = a.
\]

\begin{notation}
    We may extend this notation for
    \begin{itemize}
        \item \(a^0 = e\),
        \item \(a^n = a^{n - 1} \cdot a\) for any \(n \in \NN\).
    \end{itemize}

    Similarly, for negative integers \(n < 0\), we define
    \[
        a^n = \para{a \inverse}^{-n}.
    \]
\end{notation}

We may show that \(a^m \cdot a^n = a^{m + n}\) and \(\para{a^m}^n = a^{mn}\), that is, the exponentiation satisfies the rules we are familiar with.

Recall that it is \emph{not} necessarily true that \(a \cdot b = b \cdot a\) for all \(a, b \in G\).

Hence, it is \emph{not} necessarily true that
\[
    \para{a \cdot b}\inverse = a \inverse \cdot b \inverse.
\]

However, we notice that
\begin{align*}
    \para{a \cdot b} \cdot \para{b \inverse \cdot a \inverse} &= \para{\para{a \cdot b} \cdot b\inverse} \cdot a\inverse \\
    &= \para{a \cdot \para{b \cdot b\inverse}} \cdot a\inverse \\
    &= \para{a \cdot e} \cdot a\inverse \\
    &= a \cdot a\inverse \\
    &= e.
\end{align*}

Hence, we have that
\[
    \para{a \cdot b}\inverse = b \inverse \cdot a \inverse.
\]

We give a special name for those groups which are commutative.

\begin{definition}[Commutativity, Abelian Group]
    \label{def:abelian-group}%
    If \((G, \cdot, e)\) is a group, and
    \[
        a \cdot b = b \cdot a
    \]
    for all \(a, b \in G\), then \(G\) is \emph{commutative}, and \(G\) is called an \emph{abelian group}.
\end{definition}

\begin{example}[Trivial Group]
    If \(G = \brce{e}\) and \(\cdot: \para{e, e} \mapsto e\), then \(\para{G, \cdot, e}\) is the \emph{trivial group}.
\end{example}

\begin{examples}[Familiar Examples from Arithemetic]
    \begin{enumerate}
        \item \(\para{\ZZ, +, 0}\), \(\para{\QQ, +, 0}\), \(\para{\RR, +, 0}\) and \(\para{\CC, +, 0}\) are all \emph{abelian} groups.
        
        The inverse of \(x\) in these groups are \(-x\) -- this is known as the \emph{additive notation}.

        \item \(\para{\NN, +, 0}\) is \emph{not} a group, since it does not satisfy the Inverse Axiom.
        
        \item \(\para{\QQ, \times, 1}\) is \emph{not} a group because \(0 \in \QQ\) has no inverse. However, \(\para{\QQ \setminus \brce{0}, \times, 1}\) is an \emph{abelian group}, and we denote this as \(\QQ\multiplicative\).
        
        Similarly, \(\para{\RR \setminus \brce{0}, \times, 1}\) and \(\para{\CC \setminus \brce{0}, \times, 1}\) are as well, and we denote them as \(\RR\multiplicative\) and \(\CC\multiplicative\) respectively.
    \end{enumerate}
\end{examples}

\subsection{Definitions on Finite Groups}
\label{sec:def-finite-groups}

Most examples of groups given above are infinite, apart from the trivial group.

We would like to define the size of a group, and the definition is as in 

\begin{definition}[Order of a Group, Finite Group]
    \label{def:order-group}%
    The \emph{order} of a group \(\para{G, \cdot, e}\) is the number of elements of \(G\), denoted by \(\abs{G}\) or \(\size G\).

    If \(\abs{G} < \infinity\), then we say \(\para{G, \cdot, e}\) is a \emph{finite} group.
\end{definition}

\begin{examples}
    \begin{enumerate}
        \item For any \(n \in \NN\), let \(C_n = \set{z \in \CC}{z^n = 1}\), then \(\para{C_n, \times, 1}\) is an abelian group.
        \item Let \(\ZZ_n = \brce{0, 1, \cdots, n - 1}\). We let for \(a, b \in \ZZ_n\), we have
        \[
            a +_n b = (a + b) \modulo n.
        \]

        We can see \(\para{\ZZ_n, +_n, 0}\) is an abelian group.
    \end{enumerate}
\end{examples}

\subsection{Symmetric Groups}
\label{sec:symmetric-groups}

Symmetries on a set (with no additional constraints) are naturally interpreted as permutations, and we give a formal definition below.

\begin{definition}[Bijection, Permutation]
    \label{def:bijection-permutation}%
    Let \(X, Y\) be sets. A \emph{bijection} is a map \(X \to Y\) that has an inverse \(g: Y \to X\) such that
    \[
        f \circ g = \identity_Y, g \circ f = \identity_X
    \]
    where
    \[
        \identity_S: S \to S, x \mapsto x
    \]
    for all \(x \in S\) is the \emph{identity map}.

    Here,
    \[
        \para{f \circ g}\para{x} = f \para{g \para{x}}
    \]
    by definition of composition.

    A bijection \(X \to X\) is called a \emph{permutation}.
\end{definition}

\begin{definition}[Symmetry Group]
    \label{def:symmetry-group}%
    \(\Sym\para{X}\) is defined to be the set of permutations of a set \(X\).
\end{definition}

To show \(\Sym\para{X}\) is indeed a group, we first show \autoref{lem:composition-associative}.

\begin{lemma}[Composition is Associative]
    \label{lem:composition-associative}%
    Consider maps of sets
    \[
        f: W \to X, g: X \to Y, h: Y \to Z,
    \]
    we have
    \[
        \para{h \circ g} \circ f = h \circ \para{g \circ f}.
    \]
\end{lemma}

\begin{proof}
    For any \(w \in W\), we have
    \begin{align*}
        \para{\para{h \circ g} \circ f} \para{w} &= \para{h \circ g} \para{f\para{w}} \\
        &= h \para{g \para{f \para{w}}} \\
        &= h \para{\para{g \circ f} \para{w}} \\
        &= \para{h \circ \para{g \circ f}} \para{w}
    \end{align*}
    as required.
\end{proof}

\begin{proposition}
    \label{prop:symmetry-group}%
    For any set \(X\), \(\para{\Sym\para{X}, \circ, \identity_X}\) is a group.
\end{proposition}

\begin{proof}
    Closure is automatic, and associativity follows from \autoref{lem:composition-associative}.

    The identity on \(X\), \(\identity_X\) provides an identity element.

    Inverses exist by definition.
\end{proof}

\begin{definition}[\(S_n\)]
    \label{def:s-n}%
    If \(X = \brce{1, 2, \ldots, n}\), we write \(S_n = \Sym\para{X}\).
\end{definition}

\begin{examples}
    \begin{itemize}
        \item \(S_3\) may represent the groups to rearrange three flower pits.
        \item \(S_{52}\) may represent the group of ways to shuffle a deck of cards.
    \end{itemize}
\end{examples}

\begin{remark}
    We note that \(\abs{S_n} = n!\) by basic combinatorics.
\end{remark}

\begin{notation}
    Writing \(\para{G, \cdot, e}\) for a group is often cumbersome. Henceforth, we will normally just write \(G\) for the group.
    
    If we would like to emphasise that \(e\) is the identity element of \(G\), we will write \(e_G\) for \(e\).

    Likewise, we can write \(\cdot_G\) for the operation on \(G\).
\end{notation}

\subsection{Subgroups}
\label{sec:subgroups}

Sometimes we want to restrict our attention to smaller groups rather than the whole group, for instance, \(\ZZ\) inside \(\RR\), or the rotations that preserve a triangle, instead of all symmetries of the plane.

\begin{definition}[Subgroup]
    \label{def:subgroup}%
    Let \(G\) be a group, and \(H \subseteq G\). If we have
    \begin{itemize}
        \item \emph{Identity.} \(e \in H\);
        \item \emph{Closure.} \(a \cdot b \in H\) for all \(a, b \in H\);
        \item \emph{Inverse.} \(a \in H\) implies \(a\inverse \in H\);
    \end{itemize}
    we say \(H\) is a \emph{subgroup} of \(G\), and we write \(H \subgroup G\).
\end{definition}

\begin{remark}
    If \(G\) is a group, then any subgroup \(H \subgroup G\) is itself a group. Note associativity is not required -- it is inherited from \(G\).
\end{remark}

\begin{examples}
    \begin{enumerate}
        \item Every group \(G\) is a subgroup of itself.

        \item For any group \(G\),
        \[
            \trivialgp_G = \brce{e_G}
        \]
        is the \emph{trivial subgroup}.

        \item \(\ZZ \subseteq \QQ \subseteq \RR \subseteq \CC\), and in fact
        \[
            \ZZ \subgroup \QQ \subgroup \RR \subgroup \ZZ.
        \]

        \item For \(n = 0, 1, 2, 3, \cdots,\) let
        \[
            n\ZZ = \set{nk}{k \in \ZZ} = \brce{\cdots, -2n, -n, 0, n, 2, \cdots}.
        \]

        Now, we have
        \begin{itemize}
            \item \(0 \in \ZZ\).
            \item Let \(x = nk, y = nl \in n\ZZ\) for \(k, l \in \ZZ\).
            
            Then \(x + y = nk + nl = n \para{k + l} \in n\ZZ\).
            \item Let \(x = nk \in n\ZZ\) for \(k \in \ZZ\).
            
            Then \(-x = -nk = n \cdot \para{-k} \in n\ZZ\).
        \end{itemize}

        Hence, \(n\ZZ \subgroup \ZZ\).

        We will see in \autoref{prop:subgroups-of-integers} that all subgroups of \(\ZZ\) are of this form.
    \end{enumerate}
\end{examples}

\begin{remark}
    All subgroups of a group \(G\) that are not trivial (\(\trivialgp_G\)) or the group \(G\) itself are referred to as its \emph{proper subgroups}.
\end{remark}

\begin{proposition}
    \label{prop:subgroups-of-integers}%
    If \(H \subgroup \ZZ\), then \(H = n\ZZ\) for some \(n = 0, 1, 2, \cdots\).
\end{proposition}

\begin{proof}
    If \(H = \brce{0}\) then \(H = 0\ZZ\) is indeed of this form.

    Otherwise, \(H \neq \brce{0}\), we may choose \(n \in H \setminus \brce{0}\) to be the smallest positive \(n \in H\).

    Note that if \(x \in H\) and \(x < 0\), then \(-x \in H\) and \(x > 0\). Therefore, such positive number \(n\) must exist.

    By induction, we may show that \(nk \in H\) for all \(k = 1, 2, 3, \cdots\), and \(-n \in H\), so \(-nk \in H\) for all \(k = 1, 2, 3, \cdots\).

    This gives us \(n\ZZ \subseteq H\), and we next show \(n\ZZ = H\) by contradiction. Suppose \(n \ZZ \neq H\), so there is some \(x \in H\) such that \(x \notin n\ZZ\).

    Dividing by \(n\) and taking the remainder (integer division), we get \(x = nq + r\) for some integers \(q, r \in \ZZ\), where \(0 < r < n\).

    But we know \(r = x - nq\) and \(x \in H, nq \in n\ZZ \subseteq H\) so \(r \in H\), and this contradicts that \(n\) is the smallest.

    So \(n\ZZ = H\) as desired.
\end{proof}

We give more examples of subgroups here.

\begin{examples}[More Examples of Subgroups]
    \begin{enumerate}
        \item If \(H, K \subgroup G\), then \(H \cap K \subgroup G\). (See Example Sheet 1, Question 2).
        \item Similarly, for any family of subgroups,
        \[
            H_i \subgroup G
        \]
        for \(i \in I\), an arbitrary index set (could be uncountable). We have
        \[
            \bigcap_{i \in I} H_i = \set{a \in G}{\forall i \in I: a \in H_i} \subgroup G.
        \]
    \end{enumerate}
\end{examples}

Specifically, the second example allows us to define a 'smallest group' containing a set as follows.

\begin{definition}[Subgroup Generated by a Set]
    \label{def:generating-set}%
    Let \(X\) be a subset of a group \(G\). Then we define
    \[
        \ang{X} = \bigcup_{X \subseteq H \subgroup G} H
    \]
    is the smallest subgroup of \(G\) containing \(X\).

    It is the intersection of all subgroups of \(G\) that contains \(X\), and this is called \emph{the subgroup generated by \(X\)}.

    If \(G = \ang{X}\) we say \(X\) \emph{generates} \(G\), or \(X\) is a \emph{generating set} for \(G\).
\end{definition}

\begin{remark}
    Intuitively, \(\ang{X}\) is the \emph{smallest} subgroup containing \(X\), and if \(X\) generates \(G\), this means every element in \(g \in G\) can be written as
    \[
        g = x_1^{\pm 1} \cdot x_2^{\pm 1} \cdots x_n^{\pm 1}
    \]
    for some \(n \geq 0\), where all \(x_i \in X\), not necessarily distinct. (See Example Sheet 1 Q9 (a).)
\end{remark}

\subsection{Geometric Examples}
\label{sec:geometric-examples}

Let \(\CC\) be the complex plane, equipped with the usual notion of distance (the Euclidean norm), where
\[
    \abs{z - w} = \sqrt{\para{z_1 - w_1}^2 + \para{z_2 - w_2}^2}.
\]

\begin{figure}[ht!]
    \centering
    \caption{Distance on \(\CC\)}
    \label{fig:distance-on-C}
\end{figure}

We consider a certain group of symmetries that preserves distances.

\begin{definition}[Isometry]
    \label{def:isometry}%
    For any subset \(X \subseteq \CC\), an \emph{isometry} of \(X\) is a bijection \(f: X \to X\) that preserves distance, that is,
    \[
        \abs{f\para{x} - f\para{y}} = \abs{x - y}
    \]
    for all \(x, y \in X\).
\end{definition}

\begin{proposition}[Isometry Groups in \(\CC\)]
    \label{prop:isometry-group-in-C}%
    Let \(X \subseteq \CC\). The set of isometries of \(X\), denoted as \(\Isom\para{X}\), is a subgroup of \(\Sym\para{X}\):
    \[
        \Isom\para{X} \subgroup \Sym\para{X}.
    \]

    In particular, \(\Isom\para{X}\) is a group.
\end{proposition}

\begin{proof}
    We check \autoref{def:subgroup} (the definition of a subgroup). Let \(f, h \in \Isom\para{X}\), and \(x, y \in X\).

    \begin{itemize}
        \item We see that \(\identity_{X} \in \Isom\para{X}\) since \(\identity_{X}\) is a bijection \(X \to X\), and
        \[
            \abs{\identity\para{x} - \identity\para{y}} = \abs{x - y},
        \]
        so it is indeed an isometry.

        \item Since \(f, g\) are isometries,
        \begin{align*}
            \abs{f\para{g \para{x}} - f\para{g \para{y}}} &= \abs{g\para{x} - g\para{y}} && f \in \Isom\para{X} \\
            &= \abs{x - y}, && g \in \Isom\para{X}
        \end{align*}
        so \(f \circ h \in \Isom\para{X}\).

        \item If \(f \in \Isom\para{X}\), let \(x' = f\inverse\para{x}\) and \(y' = f\inverse\para{y}\), then
        \begin{align*}
            \abs{x' - y'} &= \abs{f\para{x'} - f\para{y'}} && f \in \Isom\para{X} \\
            &= \abs{f \circ f\inverse \para{x} - f \circ f\inverse \para{y}} \\
            &= \abs{\identity_{X} \para{x} - \identity_{X} \para{y}} \\
            &= \abs{x - y},
        \end{align*}
        so \(f \inverse \in \Isom\para{X}\), as desired.
    \end{itemize}
\end{proof}

\begin{definition}[Dihedral Group]
    Let \(X_n \subseteq \CC\) be the \(n\)-gon with vertices
    \[
        \set{e^{\frac{2 \pi i}{n}}}{k = 0, 1, \ldots, n - 1}
    \]
    for \(n \geq 3\), as in \autoref{fig:regular-n-gon-on-C}.

    The \emph{\(n\)-th Dihedral Group} \(D_{2n}\) is defined as
    \[
        D_{2n} = \Isom\para{X_n}.
    \]
\end{definition}

\begin{figure}[ht!]
    \centering
    \caption{The regular \(n\)-gon on \(\CC\)}
    \label{fig:regular-n-gon-on-C}
\end{figure}

\begin{example}[\(D_6\)]
    \(D_{2 \cdot 3} = D_6\) is the symmetry group of the equilateral triangle that we studied at the start.
\end{example}

\begin{theorem}[Size of Dihedral Group]
    \label{thm:size-of-dihedral-group}%
    \(\abs{D_{2n}} = 2n\).
\end{theorem}

The proof will give us a good description of all the elements in \(D_{2n}\). However, we first provide some geometric lemmas, the first of which is a basic fact from Euclidean geometry.

\begin{lemma}[Kite Lemma]
    \label{lem:kite-lemma}%
    Let \(x_1, y_1, x_2, y_2 \in \CC\). If \(\abs{x_1 - y_1} = \abs{x_1 - y_2}\), and \(\abs{x_2 - y_1} = \abs{x_2 - y_2}\), then
    \[
        \para{x_2 - x_1} \perp \para{y_2 - y_1}.
    \]
\end{lemma}

\begin{figure}[ht!]
    \centering
    \caption{The Kite Lemma}
    \label{fig:kite-lemma}
\end{figure}

\begin{proof}
    Referring to \autoref{fig:kite-lemma}, let \(z\) be the intersection of \(x_1 x_2\) and \(y_1 y_2\). By symmetry, we have \(\angle x_1 z y_1 = \angle x_1 z y_2\).

    However, \(y_1 y_2\) is a straight line, so \(\angle x_1 z y_1 + \angle x_1 z y_2 = \pi\), and therefore \(\angle x_1 z y_1 = \frac{\pi}{2}\), as desired.
\end{proof}

\begin{lemma}[Three-Point Lemma]
    \label{lem:three-point-lemma}%
    Let \(X \subseteq \CC\) and \(f \in \Isom\para{X}\). If there exist some \(x_1, x_2, x_3 \in X\) that are not co-linear, such that
    \[
        x_i = f\para{x_i}
    \]
    for \(i = 1, 2, 3\), then \(f = \identity_{X}\).
\end{lemma}

\begin{proof}
    We prove by contradiction. Suppose that \(f\para{y} \neq y\) for some \(y \in X\), then
    \[
        \abs{f\para{y} - x_i} = \abs{f\para{y} - f\para{x_i}} = \abs{y - x_i}
    \]
    by the hypothesis and that \(f \in \Isom\para{X}\), for \(i = 1, 2, 3\).

    We apply \autoref{lem:kite-lemma} (the Kite Lemma) with \(y_1 = y, y_2 = f\para{y}\), and hence \(\para{x_2 - x_1}\) is perpendicular to \(\para{f\para{y} - y} \neq 0\).

    Similarly, \(\para{x_3 - x_1}\) is perpendicular to \(\para{f\para{y} - y} \neq 0\).

    Hence, \(x_1, x_2, x_3\) lie on a line, which contradicts with \(x_1, x_2, x_3\) being not co-linear.
\end{proof}

\begin{remark}
    There is equally a \(\para{n + 1}\)-point lemma valid in \(\RR^n\) equipped with the usual notion of distance.
\end{remark}

Now, to prove \autoref{thm:size-of-dihedral-group} (that \(\abs{D_{2n}} = 2n\)), we define \emph{2 elements} of \(D_{2n}\):
\begin{align*}
    r\para{z} &= e^{\frac{2 \pi i}{n}} z, && \text{a rotation} \\
    s\para{z} &= \conjugate{z}. && \text{a reflection} \\
\end{align*}

\begin{figure}[ht!]
    \centering
    \caption{Two Elements of \(D_{2n}\)}
    \label{fig:two-elements-of-d-2n}
\end{figure}

We claim that in particular,
\[
    D_{2n} = \brce{e, r, r^2, \ldots, r^{n - 1}, s, rs, \ldots, r^{n - 1} s}
\]
and that these two elements generate the dihedral group,
\[
    D_{2n} = \ang{r, s}.
\]

The proof structure would be a typical subset-superset proof, i.e.
\begin{enumerate}
    \item \(\brce{e, r, r^2, \ldots, r^{n - 1}, s, rs, \ldots, r^{n - 1} s} \subseteq D_{2n}\);
    \item \(D_{2n} \subseteq \brce{e, r, r^2, \ldots, r^{n - 1}, s, rs, \ldots, r^{n - 1} s}\).
\end{enumerate}

\begin{proof}
    First, we show that \(r, s \in D_{2n}\). Indeed, for any \(x, y \in \CC\),
    \begin{align*}
        \abs{r\para{x} - r\para{y}} &= \abs{e^{\frac{2 \pi i}{n}} x - e^{\frac{2 \pi i}{n}} y} \\
        &= \abs{e^{\frac{2 \pi i}{n}}} \abs{x - y} \\
        &= \abs{X - y}
    \end{align*}
    so \(r\) is indeed an isometry of \(\CC\).

    Also, applying \(r\) to a particular vertex,
    \begin{align*}
        r\para{e^{\frac{2 \pi i k}{n}}} &= e^{\frac{2 \pi i}{n}} \cdot e^{\frac{2 \pi i k}{n}} \\
        &= e^{\frac{2 \pi i}{n} \para{k + 1}}
    \end{align*}
    is another vertex.

    So \(r\) sends vertices of \(X_n\) to vertices of \(X_n\), and preserves \(X_n\).

    Similarly, for any \(x, y \in \CC\),
    \begin{align*}
        \abs{s \para{x} - s \para{y}} &= \abs{\conjugate{x} - \conjugate{y}} \\
        &= \abs{\conjugate{x - y}} \\
        &= \abs{x - y}
    \end{align*}
    so \(s\) is an isometry on \(\CC\).

    Furthermore, applying \(s\) to a vertex of \(X_n\), we have
    \begin{align*}
        s\para{e^{\frac{2 \pi i k}{n}}} &= e^{- \frac{2 \pi i k}{n}} \\
        &= e^{\frac{2 \pi i \para{n - k}}{n}}
    \end{align*}
    is another vertex of \(X_n\).

    Therefore, we have shown that \(\brce{r, s} \subseteq D_{2n}\), and therefore, by induction,
    \[\brce{e, r, r^2, \ldots, r^{n - 1}, s, rs, \ldots, r^{n - 1} s} \subseteq D_{2n}.\]

    To show that indeed these elements are all the elements of \(D_{2n}\), we let \(f \in D_{2n}\) be an isometry on \(X_n\), and consider its action of three points on the polygon, as in \autoref{fig:three-points-on-polygon}.

    \begin{figure}[ht!]
        \centering
        \caption{Three Points on the Polygon}
        \label{fig:three-points-on-polygon}
    \end{figure}

    Namely, let \(x = 1, y = e^{\frac{2 \pi i}{n}}, z = e^{- \frac{2 \pi i}{n}}\).

    Since \(f \in D_{2n}\), \(f(x)\) is a vertex of \(X_n\), so
    \[
        f\para{x} = e^{\frac{2 \pi i k}{n}}
    \]
    for some \(k = 0, 1, \ldots, n - 1\).

    Therefore,
    \begin{align*}
        r^{-k} \circ f \para{x} &= r^{-k} \para{e^{\frac{2 \pi i k}{n}}} \\
        &= e^{- \frac{2 \pi i k}{n}} \cdot e^{\frac{2 \pi i k}{n}} \\
        &= 1 \\
        &= x.
    \end{align*}

    Now, \(r^{-k} \circ f \in D_{2n}\) since \(r^{-k}, f \in D_{2n}\). Therefore,
    \begin{align*}
        \abs{r^{-k} \circ f \para{y} - x} &= \abs{r^{-k} \circ f \para{y} - r^{-k} \circ f \para{x}} \\
        &= \abs{y - x},
    \end{align*}
    so \(r^{-k} \circ f \para{y} \in \brce{y, z}\), and similarly \(r^{-k} \circ f \para{z} \in \brce{y, z}\) but \(r^{-k} \circ f \para{y} \neq r^{-k} \circ f \para{z}\).

    Therefore, there are \(2\) cases here:
    \begin{enumerate}
        \item \(r^{-k} \circ f \para{y} = y, r^{-k} \circ f \para{z} = z.\) Then \(r^{-k} \circ f\) fixes \(x, y\) and \(z\), which are three non-collinear points. This means
        \[
            r^{-k} \circ f = \identity_{X_n}
        \]
        by \autoref{lem:three-point-lemma} (the three-point lemma), and hence
        \[
            f = r^k \in \brce{e, r, \ldots, r^{n - 1}} \subseteq \brce{e, r, \ldots, r^{n - 1}, s, rs, \ldots, r^{n - 1}s}.
        \]

        \item We have
        \begin{align*}
            r^{-k} \circ f \para{x} &= x = s\para{x}, \\
            r^{-k} \circ f \para{y} &= z = s\para{y}, \\
            r^{-k} \circ f \para{z} &= y = s\para{z}.
        \end{align*}

        Hence, left-multiplying both sides by \(s\inverse = s\), we have that \(s\inverse \circ r^{-k} \circ f\) fixes \(x, y, z\).

        Hence, by \autoref{lem:three-point-lemma} (the three-point lemma),
        \[
            s\inverse \circ r^{-k} \circ f = \identity_{X_n}
        \]
        and hence
        \[
            f = r^k s \in \brce{s, rs, \ldots, r^{n - 1}s} \subseteq \brce{e, r, \ldots, r^{n - 1}, s, rs, \ldots, r^{n - 1}s},
        \]
        as required.
    \end{enumerate}

    So far, we have shown that
    \[
        D_{2n} = \brce{e, r, \ldots, r^{n - 1}, s, rs, \ldots, r^{n - 1}s}.
    \]

    Now, we show that the right-hand side has size \(2n\), so we check all elements on the right-hand side are distinct from each other.

    First, if \(0 \leq k, l < n\) such that \(r^k = r^l\), then in particular, \(r^k \para{1} = r^l \para{1}\), and hence \(e^{\frac{2 \pi i k}{n}} = e^{\frac{2 \pi i l}{n}}\), and hence \(k = l\).

    If \(r^{k} = s\), then
    \[
        e^{\frac{2 \pi i k}{n}} = r^{k} \para{1} = s \para{1} = 1,
    \]
    so \(k = 0\), and \(s = r^{0} = \identity\). But then
    \[
        z = s\para{y} = \identity\para{y} = y,
    \]
    which is impossible. So \(r^{k} \neq s\) for any \(k\).

    Finally, if \(r^{k} = r^{l} s\), then \(s = r^{k - l}\), which is impossible by previous case; if \(r^{k} s = r^{l} s\), then \(r^{k} = r^{l}\), then \(k = l\).

    So indeed, we finished proving that \(\abs{D_{2n}} = 2n\).
\end{proof}

To understand the group operation on \(D_{2n}\), we need to understand the different ways to multiply \(r\) and \(s\), which leads to \autoref{lem:dihedral-relation}.

\begin{lemma}[Dihedral Relation]
    \label{lem:dihedral-relation}%
    For \(r, s \in D_{2n}\) as above, we have
    \[
        s r = r^{-1} s.
    \]
\end{lemma}

\begin{proof}
    By \autoref{lem:three-point-lemma} (the three-point lemma), it suffices to check that the two expressions do the same thing to \(x, y\) and \(z\).

    Indeed,
    \begin{align*}
        sr\para{x} &= \conjugate{e^{\frac{2 \pi i}{n}}} \\
        &= e^{-\frac{2 \pi i}{n}} \\
        &= r\inverse \para{x} \\
        &= r\inverse s \para{x},
    \end{align*}
    \begin{align*}
        sr\para{y} &= \conjugate{e^{\frac{2 \pi 2 i}{n}}} \\
        &= e^{-\frac{2 \pi 2 i}{n}} \\
        &= r\inverse \para{e^{- \frac{2 \pi i}{n}}} \\
        &= r\inverse s \para{y},
    \end{align*}
    and
    \begin{align*}
        sr\para{z} &= \conjugate{1} \\
        &= 1 \\
        &= r\inverse \para{e^{\frac{2 \pi i}{n}}} \\
        &= r\inverse s \para{z}.
    \end{align*}

    Hence, \(sr = r\inverse s\).
\end{proof}

\subsection{Homomorphisms and Isomorphisms}
\label{sec:homomorphism-isomorphism}

Some groups are not literally (set-theoretically) equal, but none-the-less having the same structure.

\begin{example}[Groups that seem equal]
    We would say that \(\Sym\para{\text{deck of cards}}\) are somehow equal to \(\Sym\para{\text{52 undergraduate students}}\) in a way, since they all represent permutations on 52 objects.
\end{example}

\begin{definition}[Homomorphism]
    \label{def:homomorphism}%
    A map between groups \(\phi: G \to H\) is called a \emph{homomorphism} if it preserves the group structure, the operation:
    \[
        \phi\para{g \cdot_{G} g'} = \phi\para{g} \cdot_{H} \phi\para{g'}
    \]
    for all \(g, g' \in G\).
\end{definition}

\begin{examples}[Examples of Homomorphisms]
    \begin{enumerate}
        \item For any two groups \(G, H\), the map
        \[
            \phi: G \to H, g \mapsto e_{H}
        \]
        is a homomorphism. This is the \emph{trivial homomorphism}.
        
        \item If \(H \subgroup G\), the map
        \[
            i: H \to G, h \mapsto h
        \]
        is a homomorphism. This is the \emph{inclusion homomorphism}.

        \item Recall \(C_n = \set{z \in \CC}{z^n = 1}\) for \(n \in \NN\). If \(n \divides m\), then the map
        \[
            \phi: C_m \to C_n, z \mapsto z^{\frac{m}{n}}
        \]
        is a homomorphism.

        \item Since \(\det \para{AB} = \det \para{A} \det \para{B}\), the determinant
        \[
            \det: \GL_2\para{\RR} \to \para{\RR \setminus \brce{0}, \times} = \RR\multiplicative, A \mapsto \det A
        \]
        is a homomorphism.
    \end{enumerate}
\end{examples}

Homomorphisms between groups must map certain elements to certain elements.

\begin{lemma}[Homomorphisms Preserve the Identity and Inverses]
    \label{lem:homomorphism-preserve-identity-inverse}%
    If \(\phi: G \to H\) is a homomorphism, then
    \begin{enumerate}
        \item \(\phi\para{e_G} = e_H\),
        \item \(\phi\para{g\inverse} = \phi\para{g}\inverse\) for all \(g \in G\).
    \end{enumerate}
\end{lemma}

\begin{proof}
    \begin{enumerate}
        \item We have
        \begin{align*}
            \phi\para{e_G} \cdot \phi\para{e_G} &= \phi\para{e_G \cdot e_G} && \phi \text{ is a homomorphism} \\
            &= \phi\para{e_G}
        \end{align*}
        and hence \(\phi\para{e_G} = e_H\) by \autoref{prop:inverse-identity} that the inverse is unique.

        \item We have
        \begin{align*}
            \phi\para{g} \cdot \phi\para{g \inverse} &= \phi\para{g \cdot g\inverse} && \phi \text{ is a homomorphism} \\
            &= \phi\para{e_G} \\
            &= e_H,
        \end{align*}
        so \(\phi\para{g \inverse} = \phi\para{g}\inverse\) by the definition of the inverse.
    \end{enumerate}
\end{proof}

\begin{definition}[Isomorphism]
    \label{def:isomorphism}%
    If a homomorphism \(\phi: G \to H\) is also a \emph{bijection} (a map with an inverse), then \(\phi\) is an \emph{isomorphism}, and we write
    \[
        G \isomorphic H. 
    \]
\end{definition}

From the perspective of group theory, \(G\) and \(H\) are \textbf{the same}.

\begin{examples}[Examples of Isomorphic Groups]
    \begin{enumerate}
        \item Recall that
        \[
            C_n = \set{z \in \CC}{z^n = 1}
        \]
        and
        \[
            \ZZ_n = \brce{0, 1, \ldots, n - 1}
        \]
        equipped with \(+_n\).

        Let
        \[
            \phi: \ZZ_n \to C_n, k \mapsto e^{\frac{2 \pi i k}{n}}.
        \]

        Clearly, \(\phi\) is bijective. Furthermore, for any \(k, l \in \ZZ_{n}\), we have
        \[
            k + l = np + \para{k +_n l}
        \]
        for some \(p \in \ZZ\).

        Hence,
        \begin{align*}
            \phi\para{k +_n l} &= e^{\frac{2 \pi i \para{k +_n l}}{n}} && \text{by definition of }\phi \\
            &= e^{\frac{2 \pi i \para{k +_n l}}{n}} \cdot e^{\frac{2 \pi i p n}{n}} \\
            &= e^{\frac{2 \pi i}{n} \para{k + l}} \\
            &= e^{\frac{2 \pi i k}{n}} e^{\frac{2 \pi i l}{n}} \\
            &= \phi{k} \cdot \phi{l}
        \end{align*}
        so \(\phi\) is indeed a homomorphism.

        So \(\phi\) is in fact an isomorphism, and
        \[
            \ZZ_{n} \isomorphic C_n
        \]
        for all \(n\).

        \item We have
        \[
            \para{\RR, +, 0} \isomorphic \para{\RR_{>0}, \times, 1}
        \]
        by considering the exponentiation function
        \[
            \exp: \RR \to \RR_{>0}, x \mapsto e^x.
        \]

        By laws on exponentiation, \(\exp\) is indeed a homomorphism, and it has \(\log\) as its inverse, so is bijective, and hence isomorphisms.
    \end{enumerate}
\end{examples}

\autoref{lem:homomorphism-isomorphism} allows us to think that isomorphic groups are the same/

\begin{lemma}[Properties on Homomorphisms and Isomorphisms]
    \label{lem:homomorphism-isomorphism}%
    \begin{enumerate}
        \item If \(\phi: G \to H\) is an isomorphism, so is \(\phi \inverse\).
        \item If
        \[
            G \overrightarrow{\phi} H \overrightarrow{\psi} K
        \]
        and \(\phi\) and \(\psi\) are both homomorphisms, then
        \[
            \psi \circ \phi: G \to K
        \]
        is also a homomorphism.
        \item \(\isomorphic\) is an equivalence relation.
    \end{enumerate}
\end{lemma}

We saw that every subgroup leads to an inclusion homomorphism. It is also true that homomorphisms give subgroups.

\begin{definition}[Image and Kernel]
    \label{def:image-kernel}%
    Let \(\phi: G \to H\) be a homomorphism.

    \begin{enumerate}
        \item The \emph{image} of \(\phi\) is
        \[
            \img \phi = \set{h \in H}{\exists g \in G: \phi\para{g} = h} \subseteq H.
        \]
        \item The \emph{kernel} of \(\phi\) is
        \[
            \ker \phi = \set{g \in G}{\phi\para{g} = e_H} \subseteq G.
        \]
    \end{enumerate}
\end{definition}

\begin{figure}[ht!]
    \centering
    \caption{Image and Kernel}
    \label{fig:image-and-kernel}
\end{figure}

\begin{proposition}[Image and Kernel are Subgroups]
    If \(\phi: G \to H\) is a homomorphism, then
    \[
        \img \phi \subgroup H, \ker \phi \subgroup G
    \]
\end{proposition}

\begin{proof}
    We verify the subgroup axioms.

    \begin{enumerate}
        \item For \(\img \phi\),
        \begin{itemize}
            \item \(e_H = \phi\para{e_G}\) and \(e_G \in G\), so \(e_H \in \img \phi\).
            \item For \(\phi\para{g_1}, \phi\para{g_2} \in \img \phi\), where \(g_1, g_2 \in G\), we have
            \[
                \phi\para{g_1} \cdot \phi\para{g_2} = \phi\para{g_1 \cdot g_2} \in \img \phi
            \]
            since \(g_1 \cdot g_2 \in G\).
            \item If \(\phi \para{g} \in \img \phi\) for some \(g \in G\),
            \[
                \phi\para{g}\inverse = \phi\para{g\inverse} \in \img \phi
            \]
            since \(g\inverse \in G\).
        \end{itemize}

        So \(\img \phi \subgroup H\).
        
        \item For \(\ker \phi\),
        \begin{itemize}
            \item \(\phi\para{e_G} = e_H\), so \(e_G \in \ker \phi\).
            \item If \(g_1, g_2 \in \ker \phi\),
            \[
                \phi\para{g_1 \cdot g_2} = \phi\para{g_1} \cdot \phi\para{g_2} = e_H \cdot e_H = e_H
            \]
            so \(g_1 \cdot g_2 \in \ker \phi\).
            \item If \(g \in \ker \phi\),
            \[
                \phi\para{g\inverse} = \phi\para{g}\inverse = e_H\inverse = e_H
            \]
            so \(g \inverse \in \ker \phi\).
        \end{itemize}

        So \(\ker \phi \subgroup G\).
    \end{enumerate}
\end{proof}

\begin{remarks}
    We can use the image and kernel to determine whether a homomorphism is an isomorphism. Let \(\phi: G \to H\) be a homomorphism.
    \begin{enumerate}
        \item \(\phi\) is \emph{surjective} if and only if \(\img \phi = H\).
        \item \(\phi\) is \emph{injective} if and only if \(\ker \phi = \trivialgp_{G}\).
    \end{enumerate}
\end{remarks}

We prove the second remark.

\begin{proof}
    If \(\phi\) is injective, then
    \[
        \ker \phi = \phi\inverse\para{e_H}
    \]
    has \(\leq 1\) element, which must be \(e_G\).

    Conversely, suppose \(\ker \phi = \trivialgp_G\). If \(\phi\para{g_1} = \phi\para{g_2}\), then
    \[
        \phi\para{g_1 g_2\inverse} = \phi\para{g_1} \phi\para{g_2}\inverse = e_H
    \]
    and hence
    \[
        g_1 g_2\inverse \in \ker \phi = \trivialgp_G.
    \]

    Therefore, \(g_1 g_2\inverse = e_G\), and \(g_1 = g_2\), therefore \(\phi\) is injective.
\end{proof}

\begin{remark}
    Combining the previous remark, we see a homomorphism \(\phi: G \to H\) is an isomorphism if and only if \(\img \phi = H\) and \(\ker \phi = \trivialgp_G\).
\end{remark}

\subsection{Cyclic Groups}
\label{sec:cyclic-groups}

The groups \(C_n \isomorphic \ZZ_n\) that we described above has a special property.

\begin{definition}[Cyclic Group]
    \label{def:cyclic-group}%
    A group \(G\) is \emph{cyclic} if there is \(g \in G\) and that
    \[
        G = \set{g^k}{k \in \ZZ} = \ang{g}.
    \]

    That is, \(\brce{g}\) generates \(G\), and \(g\) is called a \emph{generator} of \(G\).
\end{definition}

\begin{examples}[Examples of Cyclic Groups]
    \begin{enumerate}
        \item \(C_n\) is cyclic with generator \(e^{\frac{2 \pi i}{n}}\).
        \item \(\ZZ\) is cyclic with generator \(1\).
        \item \(\ZZ_n\) is cyclic since \(\ZZ_n \isomorphic C_n\).
    \end{enumerate}
\end{examples}

We will see in \autoref{thm:classification-cyclic-groups} that these are in fact all the cyclic groups (up to isomorphism).

\begin{theorem}[Classification of Cyclic Groups]
    \label{thm:classification-cyclic-groups}%
    If \(G\) is cyclic, then
    \begin{itemize}
        \item \(G \isomorphic C_n\) for some \(n \in \NN\); or
        \item \(G \isomorphic \ZZ\).
    \end{itemize}
\end{theorem}

\begin{proof}
    Let \(G\) by a cyclic group with generator \(g\), and let
    \[
        S = \set{k \in \NN}{g^k = e}
    \]
    and let
    \[
        n = \begin{cases}
            \min S, & S \neq \emptyset, \\
            \infinity, & S = \emptyset,
        \end{cases}
        \in \NN \cup \brce{\infinity}.
    \]

    \begin{itemize}
        \item If \(n = \infty\), we let
        \[
            \phi: \ZZ \to G, k \mapsto g^k,
        \]
        we would like to show that \(\phi\) is an isomorphism.

        Since
        \begin{align*}
            \phi \para{k + l} &= g^{k + l} \\
            &= g^k g^l \\
            &= \phi \para{k} \phi \para{l},
        \end{align*}
        \(\phi\) is a homomorphism. \(\phi\) is surjective since \(G\) is cyclic, and from \autoref{def:cyclic-group} (the definition of a cyclic group).

        We show \(\phi\) is injective by contradiction. Suppose \(0 \neq k \in \ker \phi\), then since \(\ker \phi \subgroup \ZZ\), we may replace \(k\) by \(-k\), so we assume \(k > 0\).

        Then \(k \in S\), so \(S \neq \emptyset\), which gives a contradiction. So \(\ker \phi = \brce{0}\), and \(\phi\) is injective.

        Therefore, \(\phi\) gives an isomorphism \(\ZZ \to G\), we have \(G \isomorphic \ZZ\).

        \item If \(n \in \NN\), we define
        \[
        \phi: \ZZ_n \to G, k \mapsto g^k.
        \]

        Since \(k + l = qn + \para{k +_n l}\) for some \(q \in \ZZ\), we have
        \begin{align*}
            \phi \para{k} \phi \para{l} &= g^k g^l \\
            &= g^{k + l} \\
            &= g^{qn + \para{k +_n l}} \\
            &= g^{k +_n l} && g^n = e \\
            &= \phi \para{k +_n l},
        \end{align*}
        so \(\phi\) is a homomorphism. To show \(\ZZ_n \isomorphic G\), it remains to show that \(\phi\) is a bijection.

        Since \(G\) is cyclic, every element can be written as \(g^k\) for some \(k \in \ZZ\). Conducting integer division of \(k\) by \(n\) gives
        \[
            k = nq + r
        \]
        for some \(q \in \ZZ\) and \(r \in \ZZ_n\). Therefore,
        \begin{align*}
            g^k &= g^{nq + r} \\
            &= \para{g^n}^q g^r \\
            &= g^r && g^n = e \\
            &= \phi\para{r}
        \end{align*}
        so \(\phi\) is surjective.

        To prove that \(\phi\) is injective, suppose \(\phi\para{k} = e\) where \(k \in \ker \phi\), for some \(k \in \ZZ_{n}\). Then \(k \in S\), so \(k = 0, 1, \ldots, n - 1\).

        But if \(k \in \NN\), then \(k \geq n\), since \(n\) is the minimal element in \(S\) for \(\NN\).

        Therefore, it follows that \(k = 0\), and hence
        \[
            \ker \phi = \brce{0}
        \]
        showing that \(\phi\) is injective.

        Hence,
        \[
            G \isomorphic \ZZ_n \isomorphic C_n
        \]
        as required.
    \end{itemize}
\end{proof}

Because of \autoref{thm:classification-cyclic-groups}, we can write \(C_n\) for \emph{any} cyclic group of order \(n\), and \(C_\infinity\) for \(\ZZ\).

\begin{definition}[Order of an Element]
    \label{def:order-of-element}%
    The subgroup generated by an element \(g \in G\)
    \[
        \ang{g} = \set{g^k}{k \in \ZZ} \subgroup G
    \]
    is cyclic, so
    \[
        \ang{g} \isomorphic C_n
    \]
    for some \(n \in \NN \cup \brce{\infty}\).

    The \emph{order} of an element \(g \in G\), is the \(n\) above, and we denote this as
    \[
        n = \ord \para{g} = o\para{g}.
    \]
\end{definition}

\subsection{Dihedral Groups Revisited}
\label{sec:dihedral-group-revisited}
\begin{remark}
    First, whenever \(x, y \in G\) satisfy \autoref{lem:dihedral-relation} (the dihedral relation), here \(G\) not necessarily being dihedral,
    \[
        xy = y x\inverse,
    \]
    then for any \(l > 0\), we have
    \begin{align*}
        y x^{-l} &= \para{y x \inverse} x^{1 - l} \\
        &= x \para{y x^{1 - l}} \\
        &= \cdots\\
        &= x^l y
    \end{align*}
    by induction on \(l\).

    If \(y^2 = e\), we also have
    \begin{align*}
        y x^l &= y \para{x^l y} y \\
        &= y \para{y x^{-l}} y \\
        &= x^{-l} y
    \end{align*}
    for \(l > 0\). Therefore, \(y x^l = x^{-l} y\) for any \(l \in \ZZ\).
\end{remark}

\begin{lemma}[Homomorphism from Dihedral Group]
    \label{lem:homomorphism-dihedral-group}
    Let \(G\) be a group and \(a, b \in G\) satisfy
    \begin{enumerate}
        \item \(a^n = e\): \(a\) is a 'rotation';
        \item \(b^2 = e\): \(b\) is a 'reflection'; and
        \item \(ab = b a\inverse\): \(a, b\) satisfy \autoref{lem:dihedral-relation} (the dihedral relation),
    \end{enumerate}
    then
    \[
    \phi \para{r^k} = a^k, \phi \para{s} = b
    \]
    defines a homomorphism
    \[
        \phi: D_{2n} \to G
    \]
    with \(r \mapsto a\) and \(s \mapsto b\).
\end{lemma}

\begin{proof}
    We check that \(\phi\) does indeed give a homomorphism by considering the different elements in \(D_{2n}\).

    \begin{enumerate}
        \item For two rotations,
        \begin{align*}
            \phi\para{r^k} \phi\para{r^l} &= a^k a^l \\
            &= a^{k + l} \\
            &= a^{k +_n l} \\
            &= \phi\para{r^{k +_n l}}.
        \end{align*}
        \item For a rotation and a reflection,
        \begin{align*}
            \phi\para{r^k} \phi\para{r^l s} &= a^k a^l b \\
            &= a^{k +_n l} n \\
            &= \phi\para{r^{k +_n l} s}.
        \end{align*}
        \item For a reflection and a rotation,
        \begin{align*}
            \phi\para{r^k s} \phi\para{r^l} &= a^k b a^l \\
            &= a^{k - l} b && \text{dihedral relation} \\
            &= \phi\para{r^{k - l} s} \\
            &= \phi\para{r^k r^{-l} s} \\
            &= \phi\para{r^k s r^l}. && \text{dihedral relation}
        \end{align*}
        \item For two reflections,
        \begin{align*}
            \phi\para{r^k s} \phi\para{r^l s} &= a^k b a^l b \\
            &= a^{k - l} b^2 \\
            &= a^{k - l} \\
            &= \phi\para{r^{k - l}} \\
            &= \phi\para{r^{k - l} s s} \\
            &= \phi\para{r^{k} s r^{l} s}.
        \end{align*}
    \end{enumerate}
\end{proof}

Finally, \autoref{prop:group-isomorphic-dihedral} gives a useful condition to show a group being isomorphic to the dihedral group.

\begin{proposition}[Groups Isomorphic to the Dihedral Group]
    \label{prop:group-isomorphic-dihedral}%
    If a group \(G\) has a generating set \(\brce{a, b}\) such that
    \begin{enumerate}
        \item \(a^n = e\),
        \item \(b^2 = 2\),
        \item \(ab = b a \inverse\),
        \item \(\abs{G} = 2n\),
    \end{enumerate}
    then \(G \isomorphic D_{2n}\).
\end{proposition}

\begin{proof}
    By \autoref{lem:homomorphism-dihedral-group}, this gives a homomorphism \(\phi: D_{2n} \to G\).

    \(\phi\) is surjective since \(\brce{a, b}\) generates \(G\). Also, since
    \[
        \abs{D_{2n}} = 2n = \abs{G},
    \]
    such a surjective map must be a bijection, so this gives an isomorphism, \(G \isomorphic D_{2n}\).
\end{proof}
